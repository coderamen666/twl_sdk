FSFile

Definition

#include <nitro/fs.h>

/* file and directory structure */
typedef  struct  FSFile
{
// private:
    struct FSFile              *next;
    void                       *userdata;
    struct FSArchive           *arc;
    u32                         stat;
    void                       *argument;
    FSResult                    error;
    OSThreadQueue               queue[1];
    union
    {
        u8                      reserved1[16];
        FSROMFATProperty        prop;
    };
    union
    {
        u8                      reserved2[24];
        FSROMFATCommandInfo     arg;
    };
} FSFile;

Description

This structure is used by file and directory access functions.
All members are used only in internal processing. Direct operation of these is prohibited.

Note

  • The internal expression of this structure may change in future enhancements. Avoid programming that depends upon sizes and members in the current package.
  • This structure internally maintains bidirectional pointers to list elements and system resources, so you cannot move or duplicate it simply by copying the values per structure. There is no guarantee for such object behavior. 
    
// Example: The behavior of file2 is not guaranteed when it has been copied in this way.
FSFile file1, file2;
FS_InitFile(&file1);
FS_OpenFileEx(&file1, "rom/file1.bin", FS_FILEMODE_R);
file2 = file1;
