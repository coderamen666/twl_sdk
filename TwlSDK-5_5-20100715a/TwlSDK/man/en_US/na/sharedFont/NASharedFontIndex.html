{
    NA_SHARED_FONT_WW_L = 0,

    NA_SHARED_FONT_MAX  = 3

    NA_SHARED_FONT_CN_L = NA_SHARED_FONT_MAX,
    NA_SHARED_FONT_CN_M = 4,
    NA_SHARED_FONT_CN_S = 5,
    NA_SHARED_FONT_KR_L = 6,
    NA_SHARED_FONT_KR_M = 7,
    NA_SHARED_FONT_KR_S = 8,
    NA_SHARED_FONT_CN_KR_MAX = 9
}
NASharedFontIndex;

Elements

NA_SHARED_FONT_WW_L Index showing internal system large font sizes for Japan, US, EU, and Australia.
NA_SHARED_FONT_WW_M Index showing internal system medium font sizes for Japan, US, EU, and Australia.
NA_SHARED_FONT_WW_S Index showing internal system small font sizes for Japan, US, EU, and Australia.
NA_SHARED_FONT_CN_L Index showing internal system large font sizes for China.
NA_SHARED_FONT_CN_M Index showing internal system medium font sizes for China.
NA_SHARED_FONT_CN_S Index showing internal system small font sizes for China.
NA_SHARED_FONT_KR_L Index showing internal system large font sizes for Korea.
NA_SHARED_FONT_KR_M Index showing internal system medium font sizes for Korea.
NA_SHARED_FONT_KR_S Index showing internal system small font sizes for Korea.

Description

This index is used by the internal font API to specify internal fonts.
